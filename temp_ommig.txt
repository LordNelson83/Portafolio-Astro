/* â”€â”€ Modal del diplom â”€â”€ */
function DiplomModal({ onClose }) {
  useEffect(() => {
    const onKey = (e) => { if (e.key === "Escape") onClose(); };
    document.addEventListener("keydown", onKey);
    document.body.style.overflow = "hidden";
    return () => {
      document.removeEventListener("keydown", onKey);
      document.body.style.overflow = "";
    };
  }, [onClose]);

  return (
    <div
      className="om-diplom-backdrop"
      onClick={onClose}
      role="dialog"
      aria-modal="true"
      aria-label="Diplom â€” Digital tillgÃ¤nglighet & inkluderande design"
    >
      <div
        className="om-diplom-modal"
        onClick={(e) => e.stopPropagation()}
      >
        <button
          className="om-diplom-close"
          onClick={onClose}
          aria-label="StÃ¤ng diplom"
        >
          Ã—
        </button>

        <div className="om-diplom-viewer">
          <iframe
            src={`${DIPLOM_PDF}#toolbar=0&navpanes=0`}
            title="Diplom â€” Digital tillgÃ¤nglighet & inkluderande design"
            className="om-diplom-frame"
          />
        </div>
      </div>
    </div>
  );
}
